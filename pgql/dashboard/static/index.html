üìö External API Documentation

üîó Swagger UI

External apps use the X-App-Api-Key header to authenticate. Each app sees only its allowed tables and respects its role (read/write).

# Check your app identity curl http://localhost:8765/api/v1/me \ -H "X-App-Api-Key: YOUR_API_KEY" # List available tables curl http://localhost:8765/api/v1/schema \ -H "X-App-Api-Key: YOUR_API_KEY" # Query with natural language curl -X POST http://localhost:8765/api/v1/query \ -H "X-App-Api-Key: YOUR_API_KEY" \ -H "Content-Type: application/json" \ -d '{"prompt": "Show me the latest 10 users"}'
# pip install requests import requests BASE = "http://localhost:8765/api/v1" HEADERS = {"X-App-Api-Key": "YOUR_API_KEY"} # Check identity me = requests.get(f"{BASE}/me", headers=HEADERS).json() print(me["role"], me["allowed_tables"]) # Natural language query result = requests.post(f"{BASE}/query", headers=HEADERS, json={"prompt": "Show me the latest 10 users"}).json() print(result["answer"])
// Node.js 18+ with fetch const BASE = "http://localhost:8765/api/v1"; const HEADERS = { "X-App-Api-Key": "YOUR_API_KEY" }; // Check identity const me = await fetch(`${BASE}/me`, { headers: HEADERS }).then(r => r.json()); console.log(me.role, me.allowed_tables); // Natural language query const result = await fetch(`${BASE}/query`, { method: "POST", headers: { ...HEADERS, "Content-Type": "application/json" }, body: JSON.stringify({ prompt: "Show me the latest 10 users" }) }).then(r => r.json()); console.log(result.answer);
